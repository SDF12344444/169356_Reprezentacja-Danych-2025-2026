SELECT last_name,
       salary,
       RANK() OVER (ORDER BY salary DESC) AS ranking
FROM employees;

SELECT last_name,
       salary,
       SUM(salary) OVER () AS suma_wszystkich
FROM employees;

CREATE TABLE SALES AS SELECT * FROM HR.SALES;
CREATE TABLE PRODUCTS AS SELECT * FROM HR.PRODUCTS;

SELECT e.last_name,
       p.product_name,
       SUM(s.price * s.quantity) OVER (
           PARTITION BY e.employee_id
           ORDER BY s.sale_date
       ) AS suma_skumulowana,
       RANK() OVER (
           ORDER BY (s.price * s.quantity) DESC
       ) AS ranking
FROM sales s
JOIN employees e ON s.employee_id = e.employee_id
JOIN products p ON s.product_id = p.product_id;

SELECT e.last_name,
       p.product_name,
       s.price,
       COUNT(*) OVER (PARTITION BY p.product_id, s.sale_date) AS liczba_transakcji,
       SUM(s.price * s.quantity) OVER (PARTITION BY p.product_id, s.sale_date) AS suma_dzienna,
       LAG(s.price) OVER (PARTITION BY p.product_id ORDER BY s.sale_date) AS poprzednia_cena,
       LEAD(s.price) OVER (PARTITION BY p.product_id ORDER BY s.sale_date) AS kolejna_cena
FROM sales s
JOIN employees e ON s.employee_id = e.employee_id
JOIN products p ON s.product_id = p.product_id;

SELECT p.product_name,
       s.price,
       SUM(s.price * s.quantity) OVER (
           PARTITION BY p.product_id, TO_CHAR(s.sale_date,'YYYY-MM')
       ) AS suma_miesiac,
       SUM(s.price * s.quantity) OVER (
           PARTITION BY p.product_id, TO_CHAR(s.sale_date,'YYYY-MM')
           ORDER BY s.sale_date
       ) AS suma_narastajaca
FROM sales s
JOIN products p ON s.product_id = p.product_id;

SELECT p.product_name,
       p.product_category,
       s1.price AS cena_2022,
       s2.price AS cena_2023,
       (s2.price - s1.price) AS roznica
FROM sales s1
JOIN sales s2
ON s1.product_id = s2.product_id
AND EXTRACT(MONTH FROM s1.sale_date)=EXTRACT(MONTH FROM s2.sale_date)
AND EXTRACT(DAY FROM s1.sale_date)=EXTRACT(DAY FROM s2.sale_date)
JOIN products p ON p.product_id = s1.product_id
WHERE EXTRACT(YEAR FROM s1.sale_date)=2022
AND EXTRACT(YEAR FROM s2.sale_date)=2023;

SELECT p.product_category,
       p.product_name,
       s.price,
       MIN(s.price) OVER (PARTITION BY p.product_category) AS min_cena,
       MAX(s.price) OVER (PARTITION BY p.product_category) AS max_cena,
       MAX(s.price) OVER (PARTITION BY p.product_category)
       - MIN(s.price) OVER (PARTITION BY p.product_category) AS roznica
FROM sales s
JOIN products p ON s.product_id = p.product_id;

SELECT p.product_name,
       s.sale_date,
       s.price,
       AVG(s.price) OVER (
           PARTITION BY p.product_id
           ORDER BY s.sale_date
           ROWS BETWEEN 1 PRECEDING AND 1 FOLLOWING
       ) AS srednia_kroczaca
FROM sales s
JOIN products p ON s.product_id = p.product_id;

SELECT p.product_name,
       p.product_category,
       s.price,
       RANK() OVER (PARTITION BY p.product_category ORDER BY s.price DESC) AS ranking,
       ROW_NUMBER() OVER (PARTITION BY p.product_category ORDER BY s.price DESC) AS numer,
       DENSE_RANK() OVER (PARTITION BY p.product_category ORDER BY s.price DESC) AS dense
FROM sales s
JOIN products p ON s.product_id = p.product_id;

SELECT e.last_name,
       p.product_name,
       SUM(s.price * s.quantity) OVER (
           PARTITION BY e.employee_id
           ORDER BY s.sale_date
       ) AS suma_narastajaca,
       RANK() OVER (ORDER BY (s.price * s.quantity) DESC) AS ranking
FROM sales s
JOIN employees e ON s.employee_id = e.employee_id
JOIN products p ON s.product_id = p.product_id;
